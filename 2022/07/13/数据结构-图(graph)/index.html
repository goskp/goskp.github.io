    1. 边和顶点关系:

    边和顶点关系:

    • 设n为顶点数,e为边或弧的条数

      • 对无向图: 0 <= e <= n(n - 1)/2

        • 有向图: 0 <= e <= n(n - 1)

          • 对于有向图,每个顶点至多有n - 1条边与其他的n - 1个顶点相连,则n个顶点至多有n(n - 1)条边,但对于无向图,每条边连接2个顶点,故最多n(n - 1) / 2

    矩阵法存储:

    //矩阵法存储有向带权图
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

const int MAX = 10;
typedef char VertexType[8];  //顶点类型
typedef int VRType;          //权值描述

typedef struct Graph {
    int arc_num;              //边数
    int vex_num;              //顶点数
    VertexType vertex[MAX];   //顶点数组
    VRType matrix[MAX][MAX];  //权值
} GRAPH, *LPGRAPH;

//找顶点的序号
int SearchIndex(LPGRAPH p_graph, VertexType x) {
    for (int i = 0; i < p_graph->vex_num; ++i) {
        if (strcmp(p_graph->vertex[i], x) == 0) {
            return i;
        }
    }
    return -1;
}

LPGRAPH CreateGraph() {
    LPGRAPH p_graph = (LPGRAPH)malloc(sizeof(GRAPH));
    printf("input arc_num,ver_num:");
    scanf("%d%d", &p_graph->arc_num, p_graph->vex_num);
    printf("input %d vex:", p_graph->vertex);
    //顶点数组
    for (int i = 0; i < p_graph->vex_num; ++i) {
        scanf("%s", p_graph->vertex[i]);
    }
    memset(p_graph->matrix, 0, sizeof(int) * 100);
    printf("input arc:");
    VertexType v1, v2;
    VRType vrt;  //权值
    for (int i = 0; i < p_graph->arc_num; ++i) {
        scanf("%s%s%d", v1, v2, &vrt);
        //定位,找到在矩阵中的行列
        int pos_row = SearchIndex(p_graph, v1);
        int pos_col = SearchIndex(p_graph, v2);
        p_graph->matrix[pos_row][pos_col] = vrt;
    }
    return p_graph;
}

//遍历图
void PrintGraph(LPGRAPH p_graph) {
    for (int i = 0; i < p_graph->vex_num; ++i) {
        printf("\t%s", p_graph->vertex[i]);
    }
    printf("\n");
    for (int i = 0; i < p_graph->vex_num; ++i) {
        printf("%s\t", p_graph->vertex[i]);
        for (int j = 0; j < p_graph->vex_num; ++j) {
            printf("%d\t", p_graph->matrix[i][j]);
        }
        printf("\n");
    }
}

LPGRAPH CreateGraphByFile(const char* file_name) {
    LPGRAPH p_graph = (LPGRAPH)malloc(sizeof(GRAPH));
    FILE* read = fopen(file_name, "r");
    fscanf(read, "%d\t%d\n", &p_graph->arc_num, &p_graph->vex_num);
    for (int i = 0; i < p_graph->vex_num; ++i) {
        fscanf(read, "%s", p_graph->vertex[i]);
    }
    fscanf(read, "\n");
    memset(p_graph->matrix, 0, sizeof(int) * 100);
    VertexType v1, v2;
    VRType vrt;  //权值
    for (int i = 0; i < p_graph->arc_num; ++i) {
        fscanf(read, "%s\t%s\t%d\n", v1, v2, &vrt);
        //定位,找到在矩阵中的行列
        int pos_row = SearchIndex(p_graph, v1);
        int pos_col = SearchIndex(p_graph, v2);
        p_graph->matrix[pos_row][pos_col] = vrt;
    }
    fclose(read);
    return p_graph;
}

int main(void) {
    LPGRAPH p_graph = CreateGraphByFile("graph.txt");
    PrintGraph(p_graph);
    return 0;
}

    邻接表存储:

    #include <stdio.h>
#include <stdlib.h>
#include <string.h>

const int MAX = 10;
typedef char VertexType[8];

typedef struct ArcNode {
    int ver_index;
    struct ArcNode* p_next;
} NODE, *LPNODE;

LPNODE CreateNode(int ver_index) {
    LPNODE p_new_node = (LPNODE)malloc(sizeof(NODE));
    p_new_node->ver_index = ver_index;
    p_new_node->p_next = NULL;
    return p_new_node;
}

void MyPushFront(NODE** p_head_node, int ver_index) {
    LPNODE p_new_node = CreateNode(ver_index);
    p_new_node->p_next = (*p_head_node);
    (*p_head_node) = p_new_node;
}

//顶点信息
typedef struct VNode {
    VertexType data;
    LPNODE first_node;
} VNODE, *LPVNODE, ARRAY[MAX];

//图
typedef struct graph {
    int arc_num;
    int vex_num;
    ARRAY vertex;  //结构体数组
} GRAPH, *LPGRAPH;

//定位
int SearchByVextex(LPGRAPH p_graph, VertexType x) {
    for (int i = 0; i < p_graph->vex_num; ++i) {
        if (strcmp(p_graph->vertex[i].data, x) == 0) {
            return i;
        }
    }
    return -1;
}

LPGRAPH CreateGraph() {
    LPGRAPH p_graph = (LPGRAPH)malloc(sizeof(GRAPH));
    printf("input arc_num,ver_num:");
    scanf("%d%d", &p_graph->arc_num, &p_graph->vex_num);
    printf("input %d vex:\n", p_graph->vex_num);
    for (int i = 0; i < p_graph->vex_num; ++i) {
        scanf("%s", p_graph->vertex[i].data);
        p_graph->vertex[i].first_node = NULL;
    }
    VertexType v1, v2;
    int pos_v1;
    int pos_v2;
    printf("input arc:\n");
    for (int i = 0; i < p_graph->arc_num; ++i) {
        scanf("%s%s", v1, v2);
        pos_v1 = SearchByVextex(p_graph, v1);
        pos_v2 = SearchByVextex(p_graph, v2);
        MyPushFront(&p_graph->vertex[pos_v1].first_node, pos_v2);
        MyPushFront(&p_graph->vertex[pos_v2].first_node, pos_v1);
    }
    return p_graph;
}

void PrintGraph(LPGRAPH p_graph) {
    for (int i = 0; i < p_graph->vex_num; ++i) {
        printf("%s\t", p_graph->vertex[i].data);
        LPNODE p_move = p_graph->vertex[i].first_node;
        while (p_move) {
            printf("%s-->", p_graph->vertex[p_move->ver_index].data);
            p_move = p_move->p_next;
        }
        printf("\n");
    }
}

int main(void) {
    LPGRAPH p_graph = CreateGraph();
    PrintGraph(p_graph);
    return 0;
}
