    #pragma once
#include <cstring>
#include <iostream>
using namespace std;
class MyString {
   public:
    MyString();
    MyString(const char* str);
    MyString(const MyString& object);
    ~MyString();
    void operator=(const MyString& object);
    MyString& operator+(const MyString& object);
    friend ostream& operator<<(ostream& out, const MyString& object);
    friend istream& operator>>(istream& in, MyString& object);

    int size() const;
    int empty() const;
    char operator[](int index);
    //迭代器部分
    char* begin();
    char* end();
    class iterator {
       public:
        void operator=(char* pmove);
        bool operator!=(char* pmove);
        iterator& operator++();
        char operator*();

       protected:
        char* pmove;
    };
    //比较部分
    bool operator==(const MyString& object);

   private:
    char* str;    //存储字符串的指针
    int curSize;  //当前长度
                  // int capacity;		//容量
};

    my_string.cpp:

    #include "my_string.h"
MyString::MyString() {
    this->curSize = 0;
    this->str = nullptr;
}

MyString::MyString(const char* str) {
    this->curSize = strlen(str);
    this->str = new char[this->curSize + 1];
    strcpy_s(this->str, this->curSize + 1, str);
}

MyString::MyString(const MyString& object) {
    this->curSize = object.curSize;
    this->str = new char[this->curSize + 1];
    strcpy_s(this->str, this->curSize + 1, object.str);
}

MyString::~MyString() {
    if (str != nullptr) {
        delete[] str;
        str = nullptr;
    }
}

void MyString::operator=(const MyString& object) {
    this->curSize = object.curSize;
    this->str = new char[this->curSize + 1];
    strcpy_s(this->str, this->curSize + 1, object.str);
}

MyString& MyString::operator+(const MyString& object) {
    // MyString* p = new MyString(strcat(this->str,object.str));
    char* temp = new char[this->curSize + object.curSize + 1];
    memset(temp, 0, this->curSize + object.curSize + 1);  //内存初始化
    strcat(temp, this->str);
    strcat(temp, object.str);
    MyString* p = new MyString(temp);
    delete[] temp;
    return *p;
}

int MyString::size() const {
    return curSize;
}

int MyString::empty() const {
    return curSize == 0;
}

char MyString::operator[](int index) {
    return this->str[index];
}

char* MyString::begin() {
    return this->str;
}

char* MyString::end() {
    return this->str + this->curSize + 1;
}

bool MyString::operator==(const MyString& object) {
    return strcmp(this->str, object.str) == 0;
}

ostream& operator<<(ostream& out, const MyString& object) {
    out << object.str;
    return out;
}
istream& operator>>(istream& in, MyString& object) {
    char temp[1024] = "";
    in >> temp;
    object = MyString(temp);
    return in;
}

void MyString::iterator::operator=(char* pmove) {
    this->pmove = pmove;
}

bool MyString::iterator::operator!=(char* pmove) {
    return this->pmove != pmove;
}

MyString::iterator& MyString::iterator::operator++() {
    this->pmove++;
    return *this;
}

char MyString::iterator::operator*() {
    return this->pmove[0];  //等效 return *pmove;
}

    for (int i = 0; i < array.size(); i++) {
        cout << array[i];
    }
    cout << endl;

    MyString info = "sdfasdfs";
    MyString::iterator it;
    for (it = info.begin(); it != info.end(); ++it) {
        cout << *it;
    }
    cout << endl;
    cout << (info == array) << endl;
    cout << (info == info) << endl;
    return 0;
}
