    //处理二叉搜索树的数据
typedef struct DataPair 
{
  int first;        //比较准则
  char second[20];  //字符串数据为例
} DATA, *LPDATA;

//二叉树的结点
typedef struct TreeNode 
{
  DATA data;
  struct TreeNode* p_left_child_node;
  struct TreeNode* p_right_child_node;
} TREENODE, *LPTREENODE;

//二叉搜索树
typedef struct BinarySearchTree 
{
  LPTREENODE p_root;  //根节点
  int tree_size;
} BST, *LPBST;

    //创建结点
LPTREENODE CreateTreeNode(DATA new_data) 
{
  LPTREENODE p_new_node = (LPTREENODE)malloc(sizeof(TREENODE));
  if (NULL == p_new_node) 
  {
    return NULL;
  }
  p_new_node->data = new_data;
  p_new_node->p_left_child_node = NULL;
  p_new_node->p_right_child_node = NULL;
  return p_new_node;
}

    二叉搜索树的插入:

    void InsertNode(LPBST p_tree, DATA new_data) 
{
  //插入的数据变成一个结点
  LPTREENODE p_new_node = CreateTreeNode(new_data);
  //找合适的位置
  LPTREENODE p_move = p_tree->p_root;
  LPTREENODE p_move_parent = NULL;  // p_move的父结点
  while (NULL != p_move) 
  {
    p_move_parent = p_move;
    if (new_data.first < p_move->data.first) 
    {
      //新数据小于p_move里的数据,往左走
      p_move = p_move->p_left_child_node;
    } 
    else if (new_data.first > p_move->data.first) 
    {
      //新数据大于p_move里的数据
      p_move = p_move->p_right_child_node;
    } 
    else 
    {
      //如果新数据等于p_move里的数据,
      //采用覆盖的方式
      strcpy(p_move->data.second, new_data.second);
      return;
    }
  }
  //分析查找的结果
  if (p_tree->p_root == NULL) 
  {
    //如果根节点为空
    p_tree->p_root = p_new_node;
  } 
  else 
  {
    //如果不为空,要考虑插在父结点的左边还是右边
    if (p_move_parent->data.first > new_data.first) 
    {
      //父结点的数据大于新结点的数据,插在父结点的左边
      p_move_parent->p_left_child_node = p_new_node;
    } 
    else 
    {
      //反之,插在父结点的右边
      p_move_parent->p_right_child_node = p_new_node;
    }
  }
}

//中序遍历
void InorderTraversal(LPTREENODE p_root) 
{
  if (NULL != p_root) {
    InorderTraversal(p_root->p_left_child_node);
    PrintCurNode(p_root);
    InorderTraversal(p_root->p_right_child_node);
  }
}

    查找树结点

    LPTREENODE SearchBST(LPBST p_tree, int first) 
{
  LPTREENODE p_move = p_tree->p_root;
  if (NULL == p_move) 
  {
    return p_move;
  } 
  else 
  {
    while (NULL != p_move && p_move->data.first != first) 
    {
      if (p_move->data.first > first) 
      {
        p_move = p_move->p_left_child_node;
      } 
      else 
      {
        p_move = p_move->p_right_child_node;
      }
    }
    return p_move;
  }
}

    二叉搜索树的删除:

    //删除结点
// 1.从当前结点左子树中找最右边放上去
// 2.从当前结点右子树中找最左边放上去
//创建一个结点替换要删除的结点
//再去删除拿上来的哪个结点
//二叉搜索树的删除
void DeleteNode(LPBST p_tree, int first) 
{
  LPTREENODE p_move = p_tree->p_root;
  LPTREENODE p_move_parent = NULL;
  while (NULL != p_move && p_move->data.first != first) 
  {
    p_move_parent = p_move;
    if (first < p_move->data.first) 
    {
      p_move = p_move->p_left_child_node;
    } 
    else if (first > p_move->data.first) 
    {
      p_move = p_move->p_right_child_node;
    } 
    else 
    {
      break;
    }
  }
  //分析结果
  if (NULL == p_move) 
  {
    printf("无此处,删除失败\n");
    return;
  } 
  else if (p_move->p_left_child_node != NULL && p_move->p_right_child_node != NULL) 
  {
    //左右子树都健在的情况
    LPTREENODE p_move_node = p_move->p_left_child_node;
    LPTREENODE p_move_node_parent = p_move;
    while (NULL != p_move_node->p_right_child_node) 
    {
      p_move_node_parent = p_move_node;
      p_move_node = p_move_node->p_right_child_node;
    }
    LPTREENODE p_new_node = CreateTreeNode(p_move_node->data);
    p_new_node->p_left_child_node = p_move->p_left_child_node;
    p_new_node->p_left_child_node = p_move->p_right_child_node;
    //分类讨论父节点是否存在
    if (NULL == p_move_parent) 
    {
      p_tree->p_root = p_new_node;
    } 
    else if (p_move == p_move_parent->p_left_child_node) 
    {
      p_move_parent->p_left_child_node = p_new_node;
    } 
    else 
    {
      p_move_parent->p_right_child_node = p_new_node;
    }
    //调整二叉树
    //改变删除指针的位置
    if (p_move_node_parent == p_move) 
    {  //调整的结点在左边
      p_move_parent = p_new_node;
    } 
    else 
    {
      p_move_parent = p_move_node_parent;
    }
    free(p_move);          //原来的结点干掉
    p_move = p_move_node;  //删除指针调整到要调节的位置
  }
  //只有一边的情况的删除
  LPTREENODE p_side_node = NULL;
  if (NULL != p_move->p_left_child_node) 
  {
    p_side_node = p_move->p_left_child_node;
  } 
  else 
  {
    p_side_node = p_move->p_right_child_node;
  }
  if (p_tree->p_root == p_move) 
  {
    p_tree->p_root = p_side_node;
  } 
  else 
  {
    if (p_move == p_move_parent->p_left_child_node) 
    {
      p_move_parent->p_left_child_node = p_side_node;
    } 
    else 
    {
      p_move_parent->p_right_child_node = p_side_node;
    }
  }
  free(p_move);
  p_tree->tree_size--;
}
