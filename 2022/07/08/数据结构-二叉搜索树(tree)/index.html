    //处理二叉搜索树的数据
typedef struct DatePair {
    int first;        //比较准备
    char second[20];  //字符串数据为例
} DATA, *LPDATA;

//二叉树结点类型
typedef struct TreeNode {
    DATA data;
    struct TreeNode* p_left_child;
    struct TreeNode* p_right_child;
} NODE, *LPNODE;

//二叉搜索树  BST
typedef struct BinarySearchTree {
    LPNODE p_root;  //用一个根节点表述整个树
    int tree_size;  //结点个数
} BST, *LPBST;

    创建结点:

    //创建结点
LPNODE CreateTreeNode(DATA data) {
    LPNODE p_new_tree_node = (LPNODE)malloc(sizeof(NODE));
    if (NULL == p_new_tree_node) {
        return NULL;
    }
    p_new_tree_node->data = data;
    p_new_tree_node->p_left_child = NULL;
    p_new_tree_node->p_right_child = NULL;
    return p_new_tree_node;
}

    二叉搜索树的插入:

    //二叉搜索树的插入
void InsertTreeNode(LPBST p_tree, DATA data) {
    LPNODE p_new_tree_node = CreateTreeNode(data);
    //找合适的位置
    LPNODE p_move = p_tree->p_root;
    LPNODE p_move_parent = NULL;
    while (NULL != p_move) {
        p_move_parent = p_move;
        if (data.first < p_move->data.first) {
            p_move = p_move->p_left_child;
        } else if (data.first > p_move->data.first) {
            p_move = p_move->p_right_child;
        } else {
            //相等采用覆盖的方式
            strcpy(p_move->data.second, data.second);
            return;
        }
    }
    //分析查找结果
    if (p_tree->p_root == NULL) {
        p_tree->p_root = p_new_tree_node;
    } else {
        //如果不为NULL,考虑插在parent左边还是右边
        if (p_move_parent->data.first > data.first) {
            p_move_parent->p_left_child = p_new_tree_node;
        } else {
            p_move_parent->p_right_child = p_new_tree_node;
        }
    }
    p_tree->tree_size++;
}

    LPNODE SearchBST(LPBST p_tree, int first) {
    LPNODE p_move = p_tree->p_root;
    while (NULL != p_move && p_move->data.first != first) {
        if (p_move->data.first > first) {
            p_move = p_move->p_left_child;
        } else {
            p_move = p_move->p_right_child;
        }
    }
    return p_move;
}

    二叉搜索树的删除:

    //二叉搜索树的删除
void DeleteNode(LPBST p_tree, int first) {
    LPNODE p_move = p_tree->p_root;
    LPNODE p_move_parent = NULL;
    while (NULL != p_move && p_move->data.first != first) {
        p_move_parent = p_move;
        if (first < p_move->data.first) {
            p_move = p_move->p_left_child;
        } else if (first > p_move->data.first) {
            p_move = p_move->p_right_child;
        } else {
            break;
        }
    }
    //分析结果
    if (NULL == p_move) {
        printf("无此处,删除失败\n");
        return;
    } else if (p_move->p_left_child != NULL && p_move->p_right_child != NULL) {
        //左右子树都健在的情况
        LPNODE p_move_node = p_move->p_left_child;
        LPNODE p_move_node_parent = p_move;
        while (NULL != p_move_node->p_right_child) {
            p_move_node_parent = p_move_node;
            p_move_node = p_move_node->p_right_child;
        }
        LPNODE p_new_node = CreateTreeNode(p_move_node->data);
        p_new_node->p_left_child = p_move->p_left_child;
        p_new_node->p_left_child = p_move->p_right_child;
        //分类讨论父节点是否存在
        if (NULL == p_move_parent) {
            p_tree->p_root = p_new_node;
        } else if (p_move == p_move_parent->p_left_child) {
            p_move_parent->p_left_child = p_new_node;
        } else {
            p_move_parent->p_right_child = p_new_node;
        }
        //调整二叉树
        //改变删除指针的位置
        if (p_move_node_parent == p_move) {  //调整的结点在左边
            p_move_parent = p_new_node;
        } else {
            p_move_parent = p_move_node_parent;
        }
        free(p_move);          //原来的结点干掉
        p_move = p_move_node;  //删除指针调整到要调节的位置
    }
    //只有一边的情况的删除
    LPNODE p_side_node = NULL;
    if (NULL != p_move->p_left_child) {
        p_side_node = p_move->p_left_child;
    } else {
        p_side_node = p_move->p_right_child;
    }
    if (p_tree->p_root == p_move) {
        p_tree->p_root = p_side_node;
    } else {
        if (p_move == p_move_parent->p_left_child) {
            p_move_parent->p_left_child = p_side_node;
        } else {
            p_move_parent->p_right_child = p_side_node;
        }
    }
    free(p_move);
    p_tree->tree_size--;
}
