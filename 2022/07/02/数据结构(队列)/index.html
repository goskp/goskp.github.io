    队列(FIFO)先进先出

    const int g_MAX = 10;  //队列大小

    入队:

    • 队尾往后走
    • 队头不动

    出队:

    • 队头往后走
    • 队尾不动

    注意:如果做了一次入元素和出元素,会出现伪溢出问题,最大元素个数就不是g_MAX了

    如何避免伪溢出:做成循环队列

    通过取余的方式让队尾回到队头

    描述队列结构:

    typedef struct { 
    int* p_queue_memory;
    int q_front;  //队头
    int q_tail;   //队尾
    int q_size;   //队列大小
} QUEUE, *LPQUEUE;

    用数组描述队列:

    //入队
//只需要将数据存到数组中
void Push(LPQUEUE p_queue, int data) {
    //判断队是否满,空
    if (p_queue->q_tail >= g_MAX || p_queue->p_queue_memory == NULL) {
        return;
    }
    p_queue->p_queue_memory[p_queue->q_tail++] = data;
    p_queue->q_size++;
}

//出队
void Pop(LPQUEUE p_queue) {
    if (p_queue->q_front < p_queue->q_tail) {
        p_queue->q_front++;
        p_queue->q_size--;
    }
}

//获取队头元素
int Front(LPQUEUE p_queue) {
    return p_queue->p_queue_memory[p_queue->q_front];
}

//判断是否为空
int Empty(LPQUEUE p_queue) {
    return p_queue->q_size == 0;
}

    数组描述循环队列:

    void Push(LPQUEUE p_queue, int data) {
    if (p_queue->q_size == g_MAX) {
        return;
    }
    p_queue->p_queue_memory[p_queue->q_tail % g_MAX] = data;
    p_queue->q_tail++;
    p_queue->q_size++;
}

//出队
void Pop(LPQUEUE p_queue) {
    if (p_queue->q_size != 0) {
        p_queue->q_front = (p_queue->q_front + 1) % g_MAX;
        p_queue->q_size--;
    }
}

//获取队头元素
int Front(LPQUEUE p_queue) {
    return p_queue->p_queue_memory[p_queue->q_front % g_MAX];
}

    • 相关操作:

    //入队
//就是无表头链表的表尾法插入
void Push(LPQUEUE p_queue, int data) {
    LPNODE p_new_node = CreateNode(data);
    if (0 == p_queue->q_size) {
        p_queue->p_front_node = p_new_node;
        p_queue->p_tail_node = p_new_node;
    } else {
        p_queue->p_tail_node->p_next = p_new_node;
        p_queue->p_tail_node = p_new_node;
    }
    p_queue->q_size++;
}

//出队,表头法删除
void Pop(LPQUEUE p_queue) {
    if (p_queue->q_size != 0) {
        LPNODE p_next_node = p_queue->p_front_node->p_next;
        free(p_queue->p_front_node);
        p_queue->p_front_node = p_next_node;
        p_queue->q_size--;
    }
}

int Front(LPQUEUE p_queue) {
    return p_queue->p_front_node->data;
}

int Empty(LPQUEUE p_queue) {
    return p_queue->q_size == 0;
}

void DeleteQueue(LPQUEUE p_queue) {
    while (!Empty(p_queue)) {
        Pop(p_queue);
    }
    free(p_queue);
    p_queue = NULL;
}

    优先队列:

    //优先队列
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

typedef int ElementType;
#define g_MAX_SIZE 10

typedef struct {
    int priority;         //工作量
    ElementType element;  //队列中的元素
} DataType;

typedef struct {
    int size;
    DataType memory[g_MAX_SIZE];
} PQUEUE, *LPQUEUE;

LPQUEUE CreateQueue() {
    LPQUEUE p_queue = (LPQUEUE)malloc(sizeof(PQUEUE));
    if (NULL == p_queue) {
        return NULL;
    }
    // p_queue = (DataType*)malloc(sizeof(DataType));
    memset(p_queue->memory, 0, g_MAX_SIZE * sizeof(DataType));
    p_queue->size = 0;
    return p_queue;
}

int Empty(LPQUEUE p_queue) {
    if (p_queue->size == 0) {
        return 1;
    } else {
        return 0;
    }
}

void Push(LPQUEUE p_queue, DataType data) {
    if (p_queue->size >= g_MAX_SIZE) {
        return;
    } else {
        p_queue->memory[p_queue->size++] = data;
    }
}

void Pop(LPQUEUE p_queue, DataType* data) {
    DataType min;
    int min_idx = 0;
    if (p_queue->size <= 0) {
        return;
    } else {
        min = p_queue->memory[0];
        for (int i = 1; i < p_queue->size; ++i) {
            //当前作业量西小于原来的
            if (p_queue->memory[i].priority < min.priority) {
                min = p_queue->memory[i];
                min_idx = i;
            }
        }
        *data = p_queue->memory[min_idx];
        //调整数组
        for (int i = min_idx + 1; i < p_queue->size; ++i) {
            p_queue->memory[i - 1] = p_queue->memory[i];
            min_idx = i;
        }
        p_queue->size--;
    }
}

int main(int argc, char** argv) {
    LPQUEUE p_queue = CreateQueue();
    DataType tmp_data;
    FILE* fp = fopen("task.txt", "r");
    while (!feof(fp)) {
        fscanf(fp, "%d\t%d\n", &tmp_data.element, &tmp_data.priority);
        Push(p_queue, tmp_data);
    }
    int i = 1;  //序号
    printf("序号\t任务\t优先级啊\n");
    while (!Empty(p_queue)) {
        Pop(p_queue, &tmp_data);
        printf("%d\t%d\t%d\n", i, tmp_data.element, tmp_data.priority);
        i++;
    }
    putchar('\n');
    return 0;
}
