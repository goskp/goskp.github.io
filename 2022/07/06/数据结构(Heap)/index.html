    系统堆: malloc, new出来的内存,可以随便拿,随便取

    人为堆: 引申出的一种特定堆,按照二叉树形式排列的一种堆

    大顶堆:

    • 数组中依次插入元素:

      • 从下标1开始依次插入元素, ++cur_size

    • 形成二叉树形式:

      • 父结点的下标等于孩子结点的一半

      • 相邻的层数相比,上面的元素要比下面的大

    描述大顶堆单一结构:

    //存10个元素, 第0个位置不存放元素
const int g_MAX = 11;

typedef struct {
    int* p_heap_data;
    int heap_size;
} HEAP, *LPHEAP;

    创建堆:

    LPHEAP CreateHeap() {
    LPHEAP p_heap = (LPHEAP)malloc(sizeof(HEAP));
    if (NULL == p_heap) {
        return NULL;
    }
    p_heap->heap_size = 0;
    p_heap->p_heap_data = (int*)malloc(sizeof(int) * g_MAX);
    return p_heap;
}

    形成二叉树:

    void CreateBigHeap(LPHEAP p_heap, int cur_pos) {
    while (cur_pos > 1) {
        int max = p_heap->p_heap_data[cur_pos];  //假设当前下标为最大元素
        int parent_index = cur_pos / 2;          //父结点的下标
        //父结点与最大元素相比,如果当前元素大,就交换
        if (max > p_heap->p_heap_data[parent_index]) {
            //父结点与当前结点元素交换
            p_heap->p_heap_data[cur_pos] = p_heap->p_heap_data[parent_index];
            p_heap->p_heap_data[parent_index] = max;
            //当前下标变为父结点
            cur_pos = parent_index;
        } else {
            break;
        }
    }
}

    向堆中插入元素:

    void InsertHeap(LPHEAP p_heap, int new_data) {
    if (p_heap->heap_size == g_MAX - 1) {
        return;
    }
    //直接插在当前下标
    p_heap->p_heap_data[++p_heap->heap_size] = new_data;
    CreateBigHeap(p_heap, p_heap->heap_size);
}

    遍历大顶堆:

    void PrintHeap(LPHEAP p_heap) {
    for (int i = 1; i <= p_heap->heap_size; ++i) {
        printf("%d\t", p_heap->p_heap_data[i]);
    }
    putchar('\n');
}

    有序出堆:

    //出堆
//先将堆顶,再找下面一层最大的依次抛出
int OutHeap(LPHEAP p_heap) {
    int max = p_heap->p_heap_data[1];  //大顶堆第一个元素就是最大的
    int cur_pos = 1;                   //当前下标
    int child_pos = cur_pos * 2;       //孩子结点序号
    while (child_pos <= p_heap->heap_size) {
        int tmp = p_heap->p_heap_data[child_pos];  //每一层的第一个
        //同一层上进行左右结点比较
        if (child_pos + 1 <= p_heap->p_heap_data &&  //左边的元素小于右边的元素
            tmp < p_heap->p_heap_data[child_pos + 1]) {
            tmp = p_heap->p_heap_data[++child_pos];
        }
        //向下去找
        p_heap->p_heap_data[cur_pos] = tmp;
        cur_pos = child_pos;
        child_pos *= 2;
    }
    p_heap->p_heap_data[cur_pos]=p_heap->p_heap_data[p_heap->heap_size];
    --p_heap->heap_size;
    return max;
}

int Size(LPHEAP p_heap) {
    return p_heap->heap_size;
}

int Empty(LPHEAP p_heap) {
    return p_heap->heap_size == 0;
}

//测试数据
int main(int argc, char** argv) {
    srand((unsigned int)time(NULL));
    LPHEAP p_heap = CreateHeap();
    for (int i = 0; i < 10; ++i) {
        InsertHeap(p_heap, rand() % 100);
    }
    printf("存储情况:");
    PrintHeap(p_heap);
    putchar('\n');
    printf("堆排序:");
    while(!Empty(p_heap)) {
        printf("%d\t",OutHeap(p_heap));
    }
    return 0;
}
