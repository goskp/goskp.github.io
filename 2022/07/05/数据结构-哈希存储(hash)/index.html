    数组描述Hash结构:

    为数据构建关键字:

    typedef struct Pair {
    int hash_address;  //为数据构建一个关键字,用来产生hash地址
    char element[20];
} DATA, *LPDATA;

    Hash结构:

    typedef struct HashTable {
    LPDATA* table;  //注意这里是二级指针
    int divisor;    //除数, 采用取余法   key%p, p就是divisor
    int hash_size;  //当前元素个数
} HASH, *LPHASH;

    创建Hash表

    LPHASH CreateHash(int divisor) {
    LPHASH p_hash_table = (LPHASH)malloc(sizeof(HASH));
    if (NULL == p_hash_table) {
        return NULL;
    }
    //初始化hash表
    p_hash_table->hash_size = 0;
    p_hash_table->divisor = divisor;
    //动态内存申请HashTable
    p_hash_table->table = (LPDATA*)malloc(sizeof(LPDATA) * p_hash_table->divisor);
    // p_hash_table[i]:一级指针
    for (int i = 0; i < p_hash_table->divisor; ++i) {
        p_hash_table->table[i] = NULL;
    }
    return p_hash_table;
}

    获取Hash地址:

    int SearchHashAddress(LPHASH p_hash_table, int data_address) {
    //因为存在哈希地址,所以我这里单独封装一个函数做这个事情
    int pos = data_address % p_hash_table->divisor;
    int cur_pos = pos;  //查找结束的结果
    do {
        //没有值就可以存到里面
        if (p_hash_table->table[cur_pos] == NULL ||
            p_hash_table->table[cur_pos]->hash_address == data_address) {
            return cur_pos;
        }
        cur_pos = (cur_pos + 1) % p_hash_table->divisor;
    } while (cur_pos != pos);
    return cur_pos;
}

    向Hash表中插入数据:

    void InsertData(LPHASH p_hash_table, DATA data) {
    //求出哈希地址
    int pos = SearchHashAddress(p_hash_table, data.hash_address);
    if (p_hash_table->table[pos] == NULL) {
        p_hash_table->table[pos] = (LPDATA)malloc(sizeof(DATA));
        memcpy(p_hash_table->table[pos], &data, sizeof(DATA));
        p_hash_table->hash_size++;
    } else {
        //有元素
        if (p_hash_table->table[pos]->hash_address == data.hash_address) {
            //相同关键字采用覆盖方式
            strcpy(p_hash_table->table[pos]->element, data.element);
        } else {
            printf("空间已满,无法插入\n");
            return;
        }
    }
}

    遍历Hash表与测试数据:

    void PrintHashTable(LPHASH p_hash_table) {
    for (int i = 0; i < p_hash_table->divisor; ++i) {
        if (p_hash_table->table[i] == NULL) {
            printf("NULL\n");
        } else {
            printf("%d:%s\n", p_hash_table->table[i]->hash_address,
                   p_hash_table->table[i]->element);
        }
    }
}

//测试数据
int main(int argc, char** argv) {
    LPHASH p_hash_table = CreateHash(10);
    DATA array[5] = {1,"刘亦菲",12,"王菲", 11,"王冰冰",15,"Baby", 1,"睿睿"};
    for (int i = 0; i < 5; ++i) {
        InsertData(p_hash_table, array[i]);
    }
    PrintHashTable(p_hash_table);
    return 0;
}
