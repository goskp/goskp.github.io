    查找算法

    基本查找

    • find: 区间查找
    • find_if: 条件查找
    • find_first_of: 查找区间第一次出现值
    • adjacent_find:查找第一次重复的值
    • search: 子序列查找
    • search_n: 子序列查找出现的次数

    统计查找

    • count：区间统计
    • count_if: 条件统计
    • equal: 比较

    有序查找

    • binary_search: 二分查找
    • upper_bound: 查找最后一个大于查找值
    • lower_bound: 大于等于查找值
    • eqaul_range：区间比较
    #include <algorithm>
#include <functional>
#include <iomanip>
#include <iostream>
#include <string>
#include <vector>
using namespace std;
class MM {
   public:
    MM(string name = "", int score = 0) : name(name), score(score) {}
    int getScore() const { 
        return score; 
    }

   private:
    string name;
    int score;
};
class System {
   public:
    void insertData(const MM& mm) { 
        mmData.push_back(mm); 
    }
    int getOK() {
        return count_if(mmData.begin(), mmData.end(),
                        [](const MM& mm) { return mm.getScore() >= 60; });
    }
    int getNo() {
        return count_if(mmData.begin(), mmData.end(),
                        [](const MM& mm) { return mm.getScore() < 60; });
    }

   private:
    vector<MM> mmData;
};

void testUserData() {
    System* p = new System;
    p->insertData(MM("name1", 59));
    p->insertData(MM("name2", 92));
    p->insertData(MM("name3", 88));
    p->insertData(MM("name4", 54));
    p->insertData(MM("name5", 25));
    cout << "及格人数:" << p->getOK() << endl;
    cout << "不及格人数:" << p->getNo() << endl;
}
int main(int argc, char** argv) {
    int array[10] = {1, 3, 5, 7, 9, 2, 4, 6, 8, 10};
    vector<int> testData;
    testData.assign(array, array + 10);
    // 1.区间查找
    auto it = find(testData.begin(), testData.end(), 3);
    if (it != testData.end())
        cout << *it << endl;
    // 2.条件查找
    it = find_if(testData.begin(), testData.end(),
                 [](int data) { return data > 8; });
    cout << *it << endl;
    // 3.查找第一次出现的
    int temp[] = {15, 10, 6};
    cout << *find_first_of(testData.begin(), testData.end(), temp, temp + 3)
         << endl;
    // 4.查找第一次重复的元素
    vector<int> value = {1, 2, 3, 3, 2, 4, 4, 9, 9, 5, 5, 9, 9, 9};
    cout << *adjacent_find(value.begin(), value.end()) << endl;
    // 5.区间查找
    vector<int> num = {1, 2, 3};
    cout << *search(value.begin(), value.end(), num.begin(), num.end()) << endl;
    string str = {"ILoveyouIMissyou"};
    string str2 = {"you"};
    cout << *search(str.begin(), str.end(), str2.begin(), str2.end()) << endl;
    // 6.查找出现次数
    cout << *search_n(value.begin(), value.end(), 3, 9) << endl;
    // 7.统计出现次数
    cout << count(value.begin(), value.end(), 2) << endl;
    // 8.条件统计出现次数
    cout << count_if(value.begin(), value.end(), [](int i) { return i >= 6; })
         << endl;
    // 9.相等比较
    cout << boolalpha
         << equal(value.begin(), value.end(), num.begin(), num.end()) << endl;
    // 10.二分查找
    vector<int> order = {1, 2, 3, 4, 5, 6, 7, 8, 9};
    cout << boolalpha << binary_search(order.begin(), order.end(), 5) << endl;
    // 11.大于查找
    cout << *upper_bound(order.begin(), order.end(), 7) << endl;
    // 12.大于等于
    cout << *lower_bound(order.begin(), order.end(), 7) << endl;
    // 13. 大于等于的综合体: first:等于的值，second存大于值
    cout << *equal_range(order.begin(), order.end(), 7).second << endl;
    cout << *equal_range(order.begin(), order.end(), 7).first << endl;
    testUserData();
    return 0;
}

    排序和通用算法

    • sort：排序
    • stable_sort: 保持相对顺序的排序
    • merge:归并排序，不会改变原容器数据
    • inplace_merge: 归并排序，直接作用在原容器上
    • nth_element:关键字排序
    • partition: 分类处理 (按特定条件把数据分为两堆)
    • stable_partion: 保持数据相对顺序，分类处理
    • partial_sort:局部排序
    • partial_sort_copy: 局部排序结果放到新容器中
    • random_shuffle: 乱序算法
    • reverse: 逆序
    • reverse_copy: 逆序结果另存
    • rotate: 移动元素到末位
    • rotate_copy: 移位后的结果另存
    #include <algorithm>
#include <functional>
#include <iostream>
#include <list>
#include <string>
#include <vector>
using namespace std;
int main(int argc, char** argv) {
    // 1.基本排序
    vector<int> vec = {1, 3, 5, 7, 9, 0, 2, 4, 6, 8};
    sort(vec.begin(), vec.end());
    sort(vec.begin(), vec.end(), less<int>{});
    copy(vec.begin(), vec.end(), ostream_iterator<int>(cout, " "));
    cout << endl;
    sort(vec.begin(), vec.end(), greater<int>{});
    copy(vec.begin(), vec.end(), ostream_iterator<int>(cout, " "));
    cout << endl;
    // list有内置的sort算法，不能直接用sort算法
    list<int> mylist = {1, 2, 3, 4, 9, 8};
    mylist.sort();
    copy(mylist.begin(), mylist.end(), ostream_iterator<int>(cout, " "));
    cout << endl;
    // 2.保持相对顺序的排序
    vector<double> dNum = {1.11, 2.22, 3.33, 4.44, 1.11, 1.10, 4.6, 4.5};
    stable_sort(dNum.begin(), dNum.end(),
                [](double a, double b) { return int(a) < int(b); });
    copy(dNum.begin(), dNum.end(), ostream_iterator<double>(cout, " "));
    cout << endl;
    // 3.归并排序
    vector<int> order = {1, 3, 5, 7, 9, 0, 2, 4, 6, 8};
    vector<int> result(order.size());
    merge(order.begin(), order.begin() + 5, order.begin() + 5, order.end(),
          result.begin());
    copy(result.begin(), result.end(), ostream_iterator<int>(cout, " "));
    cout << endl;
    // 4.改变原容器的归并
    inplace_merge(order.begin(), order.begin() + 5, order.end());
    copy(order.begin(), order.end(), ostream_iterator<int>(cout, " "));
    cout << endl;
    // 5.关键字排序
    vector<int> element = {1, 3, 5, 7, 9, 0, 2, 4, 6, 8};
    nth_element(element.begin(), element.begin() + 2, element.end());
    copy(element.begin(), element.end(), ostream_iterator<int>(cout, " "));
    cout << endl;
    // 6.分类处理
    vector<int> pData = {1, 3, 5, 7, 9, 0, 2, 4, 6, 8};
    partition(pData.begin(), pData.end(), [](int data) { return data < 6; });
    copy(pData.begin(), pData.end(), ostream_iterator<int>(cout, " "));
    cout << endl;
    // 7.保持相对顺序分类处理
    vector<double> dNum2 = {1.11, 2.22, 3.33, 4.44, 1.11, 1.10, 4.6, 4.5};
    stable_partition(dNum2.begin(), dNum2.end(),
                     [](double a) { return int(a) < 2; });
    copy(dNum2.begin(), dNum2.end(), ostream_iterator<double>(cout, " "));
    cout << endl;
    // 8.局部排序 按元素大小顺序，选取有序的放在前面
    vector<int> testData = {19, 23, 65, 77, 45, 34};
    partial_sort(testData.begin(), testData.begin() + 3, testData.end());
    copy(testData.begin(), testData.end(), ostream_iterator<int>(cout, " "));
    cout << endl;
    // 9.局部排序结果另存
    vector<int> pResult(3);
    partial_sort_copy(testData.begin(), testData.begin() + 3, pResult.begin(),
                      pResult.end());
    copy(pResult.begin(), pResult.end(), ostream_iterator<int>(cout, " "));
    cout << endl;
    // 10.乱序算法
    srand((unsigned int)time(nullptr));
    string str = "ABCDEFG";
    random_shuffle(str.begin(), str.end());
    cout << str << endl;
    // 11.反转
    str = "ILoveyou";
    reverse(str.begin(), str.end());
    cout << str << endl;
    // 12.反转另存
    string strResult;
    strResult.resize(str.size());
    reverse_copy(str.begin(), str.end(), strResult.begin());
    cout << strResult << endl;
    // 13.移动元素到末位
    string testStr = "ILoveyou";
    rotate(testStr.begin(), testStr.begin() + 4, testStr.end());
    cout << testStr << endl;
    // 14.移位另存
    string rotateStr;
    rotateStr.resize(testStr.size());
    rotate_copy(testStr.begin(), testStr.begin() + 4, testStr.end(),
                rotateStr.begin());
    cout << rotateStr << endl;
    return 0;
}

    集合类算法

    • set_union: 并集
    • set_intersetction: 交集
    • set_difference: 差集
    • set_symmetric_difference: 对称差集
    #include <algorithm>
#include <functional>
#include <iostream>
#include <string>
#include <vector>
using namespace std;
int main(int argc, char** argv) {
    vector<int> one = {1, 2, 3, 7, 8};
    vector<int> two = {4, 5, 6, 7, 8};
    // 1.并集
    set_union(one.begin(), one.end(), two.begin(), two.end(),
              ostream_iterator<int>(cout, " "));
    cout << endl;
    // 2.交集
    set_intersection(one.begin(), one.end(), two.begin(), two.end(),
                     ostream_iterator<int>(cout, " "));
    cout << endl;
    // 3.求差集
    set_difference(one.begin(), one.end(), two.begin(), two.end(),
                   ostream_iterator<int>(cout, " "));
    cout << endl;
    // 4.对称差集 (并集减去交集)
    set_symmetric_difference(one.begin(), one.end(), two.begin(), two.end(),
                             ostream_iterator<int>(cout, " "));
    cout << endl;
    return 0;
}
