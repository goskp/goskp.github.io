    STL之list

    #include <iostream>
#include <list>
#include <string>
using namespace std;
class MM {
   public:
    MM(string name, int age, int num) : name(name), age(age), num(num) {}

   private:
    string name;
    int age;
    int num;
};

int main(int argc, char** argv) {
    //创建过程
    list<int> iList;
    list<string> sList;
    list<MM> mmData;
    //插入删除遍历查找排序
    for (int i = 0; i < 3; i++) {
        iList.push_back(i);   //尾插法
        iList.push_front(i);  //头插法
    }
    cout << "size:" << iList.size() << endl;
    cout << "empty:" << iList.empty() << endl;
    iList.sort();
    //不影响容器中元素的打印方式
    for (auto& v : iList) {
        cout << v << " ";
    }
    cout << endl;
    iList.reverse();
    list<int>::iterator it;
    for (it = iList.begin(); it != iList.end(); it++) {
        cout << *it << " ";
    }
    cout << endl;
    //一边删除一边打印
    while (!iList.empty()) {
        cout << iList.front() << " ";  //获取头元素
        iList.pop_front();             //删除头部元素
    }
    cout << "size:" << iList.size() << endl;
    //指定位置插入
    sList.push_back("ILove");
    sList.push_back("you");
    sList.insert(find(sList.begin(), sList.end(), "you"), "IMissyou");
    for (auto& v : sList) {
        cout << v << " ";
    }
    cout << endl;
    sList.erase(find(sList.begin(), sList.end(), "IMissyou"));
    for (auto& v : sList) {
        cout << v << " ";
    }
    cout << endl;
    return 0;
}

    STL之stack

    栈的存储顺序是固定顺序: FILO

    栈是不存在迭代器的

    #include <iostream>
#include <stack>
using namespace std;
int main(int argc, char** argv) {
    //后来居上的感觉
    stack<int> istack;
    for (int i = 0; i < 3; i++) {
        istack.push(i);
    }
    cout << "size:" << istack.size() << endl;
    while (!istack.empty()) {
        cout << istack.top() << " ";  //获取栈顶元素
        istack.pop();                 //出栈
    }
    cout << "size:" << istack.size() << endl;
    //悔棋，推箱子退步，寻路
    //栈解决简单问题
    //求进制
    stack<int> binary;
    int num = 999;
    cout << num << "的二进制是:";
    while (num) {
        binary.push(num % 2);
        num /= 2;
    }
    while (!binary.empty()) {
        cout << binary.top();
        binary.pop();
    }
    cout << endl;
    stack<char> hexData;
    num = 999;
    cout << num << "的十六进制是:";
    while (num) {
        if (num % 16 < 10) {
            hexData.push(num % 16 + '0');
        } else {
            hexData.push(num % 16 - 10 + 'A');
        }
        num /= 16;
    }
    while (!hexData.empty()) {
        cout << hexData.top();
        hexData.pop();
    }
    cout << endl;
    return 0;
}

    STL之queue

    队列的存取顺序也是特定，普通队列先进先出，优先队列按照优先权出队

    queue

    #include <iostream>
#include <queue>
using namespace std;
int main(int argc, char** argv) {
    queue<int> iQue;
    for (int i = 0; i < 3; i++) {
        iQue.push(i);
    }
    cout << "size:" << iQue.size() << endl;
    cout << "back:" << iQue.back() << endl;
    while (!iQue.empty()) {
        cout << iQue.front() << " ";
        iQue.pop();
    }
    cout << "size:" << iQue.size() << endl;
    return 0;
}

    priority_queue

    #include <ctime>
#include <iostream>
#include <queue>
#include <string>
#include <vector>
using namespace std;
class MM {
   public:
    MM(string name, int age) : name(name), age(age) {}
    string GetName() const {
        return name; 
    }
    int GetAge() const {
        return age;
    }
    bool operator<(const MM& object) const {
        return this->age < object.age; 
    }

   private:
    string name;
    int age;
};
//仿函数
class CompareByName {
   public:
    bool operator()(const MM& object1, const MM& object2) const {
        return object1.GetName() < object2.GetName();
    }
};

class CompareByAge {
   public:
    bool operator()(const MM& object1, const MM& object2) const {
        return object1.GetAge() < object2.GetAge();
    }
};

int main(int argc, char** argv) {
    //优先队列

    priority_queue<int, vector<int>, less<int>> lessQueue;
    priority_queue<int, vector<int>> lessQue;  //和上面是等效的
    srand((unsigned int)time(nullptr));
    for (int i = 0; i < 10; i++) {
        lessQue.push(rand() % 100);
    }
    while (!lessQue.empty()) {
        cout << lessQue.top() << " ";
        lessQue.pop();
    }
    cout << endl;
    //从小到大
    priority_queue<int, vector<int>, greater<int>> greaterQue;
    for (int i = 0; i < 10; i++) {
        greaterQue.push(rand() % 100);
    }
    while (!greaterQue.empty()) {
        cout << greaterQue.top() << " ";
        greaterQue.pop();
    }

    priority_queue<MM, vector<MM>, less<MM>> mmData;
    mmData.push(MM("baby", 19));
    mmData.push(MM("anni", 28));
    mmData.push(MM("sanshui", 21));
    cout << endl;
    while (!mmData.empty()) {
        cout << mmData.top().GetName() << "\t" << mmData.top().GetAge() << endl;
        mmData.pop();
    }

    priority_queue<MM, vector<MM>, CompareByName> mmData1;
    mmData1.push(MM("baby", 19));
    mmData1.push(MM("anni", 28));
    mmData1.push(MM("sanshui", 21));
    cout << endl;
    while (!mmData1.empty()) {
        cout << mmData1.top().GetName() << "\t" << mmData1.top().GetAge()
             << endl;
        mmData1.pop();
    }

    priority_queue<MM, vector<MM>, CompareByAge> mmData2;
    mmData2.push(MM("baby", 19));
    mmData2.push(MM("anni", 28));
    mmData2.push(MM("sanshui", 21));
    cout << endl;
    while (!mmData2.empty()) {
        cout << mmData2.top().GetName() << "\t" << mmData2.top().GetAge()
             << endl;
        mmData2.pop();
    }

    return 0;
}
