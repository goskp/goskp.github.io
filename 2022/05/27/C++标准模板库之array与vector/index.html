    STL之arary

    • array就是定长数组，任何类型定长数组
    /*
    array 创建对象的两种方式, 常对象,和指针
    size(),back(),front()方法
    下标: at(i)
    正向,逆向,迭代器遍历方法
*/
#include <array>
#include <iostream>
#include <string>
using namespace std;
int main(int argc, char* argv[]) {
    int inum[3] = {1, 2, 3};  //定长数组
    array<int, 3> number = {1, 2, 3};
    array<string, 3>* p_str = new array<string, 3>;
    (*p_str)[0] = "string1";
    (*p_str)[1] = "string2";
    (*p_str)[2] = "sdfasf";
    cout << "size:" << number.size() << endl;
    cout << "empty:" << number.empty() << endl;
    for (int i = 0; i < number.size(); i++) {
        cout << number[i] << " ";
    }
    cout << endl;
    cout << number.back() << endl;
    cout << number.front() << endl;
    for (int i = 0; i < number.size(); i++) {
        cout << number.at(i) << " ";
    }
    cout << endl;
    for (int i = 0; i < p_str->size(); i++) {
        cout << p_str->at(i) << " ";
    }
    cout << endl;
    //正向
    array<int, 3>::iterator it = number.begin();
    for (; it != number.end(); it++) {
        cout << *it << " ";
    }
    cout << endl;
    //反向
    array<int, 3>::reverse_iterator rit = number.rbegin();
    for (; rit != number.rend(); rit++) {
        cout << *rit << " ";
    }
    cout << endl;
    cout << "第一个元素:" << *number.begin() << endl;
    cout << "最后一个元素:" << *number.rbegin() << endl;
    // cout << "end:" << *number.end() << endl;  任何end都不能取值运算

    delete p_str;
    return 0;
}

    STL之vector

    • vector是动态数组
    #include <iostream>
#include <string>
#include <vector>
using namespace std;
int main(int argc, char** argv) {
    // No.1 不带长度的构建方式
    //不能直接用下标方式去操作
    vector<int> num;
    // num[0] = 1;  中断
    vector<int> number = {1, 2, 3};
    // number[3] = 333;   中断  超过原来长度
    // No.2 带长度的构造方式,在长度范围内可以直接下标法访问
    vector<string> str(3);  // 3代表数组长度
    str[0] = "sdfsd";
    str[1] = "sdfsdfsd";
    str[2] = "sdfsdfsdsd";
    // str[3] = "sdfsdfsdfsd";   //中断
    // No.3 带自动增长功能是在成员函数中完成
    // push_back(T data);
    str.push_back("sdfsdfsd");
    num.push_back(12);

    vector<int> test(3);
    test.push_back(999);
    for (auto& v : test) {
        cout << v << " ";
    }
    cout << endl;

    vector<int>::iterator it;
    for (it = test.begin(); it != test.end(); it++) {
        cout << *it << " ";
    }
    cout << endl;
    vector<int>::reverse_iterator rit;
    for (rit = test.rbegin(); rit != test.rend(); rit++) {
        cout << *rit << " ";
    }
    cout << endl;
    //成员函数
    cout << "第一个元素:" << test.front() << "\t" << *test.begin() << endl;
    cout << "最后一个二元素:" << test.back() << "\t" << *test.rbegin() << "\t"
         << *(test.end() - 1) << endl;
    cout << "最后一个元素:" << test.at(test.size() - 1) << endl;
    cout << "size:" << test.size() << endl;
    cout << "empty:" << test.empty() << endl;

    return 0;
}

    #include <array>
#include <iostream>
#include <string>
#include <vector>
using namespace std;
class MM {
   public:
    MM() {}  //如果想要构建一个无参array
    MM(string name, int age) : name(name), age(age) {}
    void printData() {
        cout << name << " " << age << endl;
    }

   private:
    string name;
    int age;
};
void initArray(array<MM, 3>& mmData) {
    mmData[0] = MM("name1", 18);
    mmData[1] = MM("name2", 29);
    mmData[2] = MM("name3", 38);
}
void printMM(array<MM, 3> mmData) {
    for (auto& v : mmData) {
        v.printData();
    }
}
void testArray() {
    array<MM, 3> mmData;
    initArray(mmData);
    printMM(mmData);
    for (int i = 0; i < mmData.size(); i++) {
        mmData[i].printData();
    }
}

void printVector(vector<MM> mmData) {
    for (auto& v : mmData) {
        // cout << v << " ";
        v.printData();
    }
}
void testVector() {
    vector<MM> mmData;
    mmData.push_back(MM("name1", 23));
    mmData.push_back(MM("name2", 25));
    mmData.push_back(MM("name3", 45));
    printVector(mmData);
}
int main(int argc, char** argv) {
    cout << "array:" << endl;
    testArray();
    cout << "vector:" << endl;
    testVector();
    return 0;
}

    互相嵌套

    #include <time.h>
#include <array>
#include <iostream>
#include <string>
#include <vector>
using namespace std;

void testArray() {
    array<array<int, 4>, 3> data;  //三乘四二维数组
    for (int i = 0; i < data.size(); i++) {
        for (int j = 0; j < data[i].size(); j++) {
            data[i][j] = j + i * data.size();  //随便赋值的
            cout << data[i][j] << " ";
        }
        cout << endl;
    }
    array<array<array<int, 4>, 4>, 4>;  //三维数组
}
void testVector() {
    vector<vector<int>> data;  //不等长的二维数组
    for (int i = 0; i < 3; i++) {
        int length = rand() % 5 + 1;
        vector<int> temp;
        for (int j = 0; j < length; j++) {
            temp.push_back(j);
        }
        data.push_back(temp);
    }
    for (int i = 0; i < data.size(); i++) {
        for (int j = 0; j < data[i].size(); j++) {
            cout << data[i][j] << " ";
        }
        cout << endl;
    }
}
void testArrayVector() {
    array<vector<int>, 3> test1;
    test1[0] = vector<int>{1, 2, 3};
    test1[1] = vector<int>{1, 2, 3, 4, 5};
    test1[2] = vector<int>{1};
    for (int i = 0; i < test1.size(); i++) {
        for (int j = 0; j < test1[i].size(); j++) {
            cout << test1[i][j] << " ";
        }
        cout << endl;
    }

    vector<array<int, 3>> test2;
    test2.push_back(array<int, 3>{1, 2, 3});
    test2.push_back(array<int, 3>{3, 3, 3});
    test2.push_back(array<int, 3>{4, 4, 4});

    for (int i = 0; i < test2.size(); i++) {
        for (int j = 0; j < test2[i].size(); j++) {
            cout << test2[i][j] << " ";
        }
        cout << endl;
    }
}
int main(int argc, char** argv) {
    srand((unsigned int)time(nullptr));
    cout << "array嵌套:" << endl;
    testArray();
    cout << "vector嵌套:" << endl;
    testVector();
    cout << "array与vector嵌套:" << endl;
    testArrayVector();

    return 0;
}

    自制array

    #include <array>
#include <iostream>
#include <string>
using namespace std;
template <class _Ty, size_t _size>
class my_array {
   public:
    _Ty& operator[](int index) { 
        return elem[index];
    }
    _Ty at(int index) { 
        return elem[index];
    }
    int size() const {
        return _size;
    }
    bool empty() const { 
        return _size == 0; 
    }
    _Ty back() {
        return elem[_size - 1]; 
    }
    _Ty front() {
        return elem[0]; 
    }
    _Ty* begin() {
        return elem; 
    }
    _Ty* end() {
        return elem + _size; 
    }
    class iterator {
       public:
        void operator=(_Ty* pmove) { 
            this->pmove = pmove; 
        }
        _Ty operator*() {
            return pmove[0]; 
        }
        bool operator!=(_Ty* pmove) {
            return this->pmove != pmove;
        }
        void operator++() { 
            this->pmove++; 
        }

       private:
        _Ty* pmove;
    };

   private:
    _Ty elem[_size];
};
int main(int argc, char** argv) {
    my_array<int, 3> data;
    data[0] = 1;
    data[1] = 2;
    data[2] = 3;
    for (auto& v : data) {
        cout << v << " ";
    }
    cout << endl;
    my_array<int, 3>::iterator it;
    for (it = data.begin(); it != data.end(); ++it) {
        cout << *it << " ";
    }
    cout << endl;
    cout << "size:" << data.size() << endl;
    cout << "empty:" << data.empty() << endl;
    array<int, 4> stl_array;
    cout << "size:" << stl_array.size() << endl;
    cout << "empty:" << stl_array.empty() << endl;
    return 0;
}
