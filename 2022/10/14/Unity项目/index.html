    • Unity3D中[SerializeField]特性的使用

      • 强制 Unity 对私有字段进行序列化

      • 当 Unity 对脚本进行序列化时，仅对公共字段进行序列化。 如果还需要 Unity 对私有字段进行序列化， 可以将 SerializeField 属性添加到这些字段

      • Unity 不会序列化 Dictionary，但您可以为键存储一个 List<> 和为值存储一个 List<>，然后在 Awake() 上将它们组合在非序列化字典中。这不能解决您需要修改字典 并将其“保存”回时出现的问题，但在许多其他情况下，这是一个方便的技巧

      • using UnityEngine;

public class SomePerson : MonoBehaviour {
    //这个字段被序列化，因为它是公共的
    public string firstName = "John";

    //这个字段不会被序列化，因为它是私有的。
    private int age = 40;

    //这个字段即使是私有的也会被序列化
    //因为它应用了SerializeField属性。
    [SerializeField]
    private bool hasHealthPotion = true;

    void Start() {
        if (hasHealthPotion) {
            Debug.Log("Person's first name: " + firstName + " Person's age: " + age);
        }
    }
}

    • HandleHorizontalMovementInput:

      • 处理玩家水平移动输入

      • 从键盘读入 ‘A’ 或者 小键盘中的向左,则将moveLeft标记为true

      • 从键盘读入 ‘D’ 或者 小键盘中的向右,则将moveRight标记为true

      private void HandleHorizontalMovementInput() {
    if (Input.GetKey(KeyCode.A) || Input.GetKey(KeyCode.LeftArrow)) {
        moveLeft = true;
    } else if (Input.GetKey(KeyCode.D) || Input.GetKey(KeyCode.RightArrow)) {
       moveRight = true;
    }
}

    • HorizontalMove(int direction):

      • 先获取玩家当前的横坐标(x)

      • 判断direction正负,正则水平右移,负责水平左移

      • 在左移右移的过程中要判定玩家是否越界

      private void HorizontalMove(int direction) {
    float playerXPos = transform.position.x;
    if (direction < 0) {
        if (playerXPos > SectionController.leftSide) {
            transform.Translate(Vector3.right * direction * moveHorizonSpeed * Time.deltaTime, Space.World);
        }
    } else if (direction > 0) {
        if (playerXPos < SectionController.rightSide) {
            transform.Translate(Vector3.right * direction * moveHorizonSpeed * Time.deltaTime, Space.World);
        }
    }
}

      • 处理玩家水平移动

      private void HandlePlayerHorizontalMovement() {
    //使玩家始终面向前方
    ResetPlayerRotation();
    //使玩家向前跑
    RunForward();
    //判断玩家左移右移
    if (moveLeft) {
        HorizontalMove(-1);
    } else if (moveRight) {
        HorizontalMove(1);
    }
}

      • 重置玩家移动状态:

      private void MovementStateReset() {
    if (Mathf.Abs(transform.position.y) <= 0.51f) {
        isGrounded = true;
    } else if (Mathf.Abs(transform.position.y) > 0.51f) {
        isGrounded = false;
    }
    moveRight = false;
    moveLeft = false;
    jumpable = false;
}
